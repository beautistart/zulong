```python
def _generate_experiences_fallback(self, analysis, draft):
    """🔥 降级：基于规则生成经验（当 L2 失败时使用）"""
    # 如果 L2 失败，使用基于规则的生成
    if analysis.get('key_decisions'):
        for decision in analysis['key_decisions']:
            experiences.append({
                'content': f'关键决策：{decision}',
                'confidence': 0.8
            })
    return experiences
```

**修改 `_generate_experiences()` 方法**:
```python
def _generate_experiences(self, analysis, draft):
    """🔥 修复：从分析结果生成经验（调用 L2 进行智能生成）"""
    try:
        # 🔥 调用 L2 进行经验生成
        experiences = self._generate_experiences_with_l2(analysis)
        
        # 如果 L2 生成失败或返回空，降级到基于规则生成
        if not experiences:
            logger.info("L2 生成经验失败，降级到基于规则生成")
            experiences = self._generate_experiences_fallback(analysis, draft)
        
        return experiences
    except Exception as e:
        logger.error(f"生成经验失败：{e}")
        return self._generate_experiences_fallback(analysis, draft)
```

### 修复 2: 完善 L2 状态管理

**修改文件**: `zulong/review/integration.py`

**修改 `_handle_quick_review()` 方法**:

```python
def _handle_quick_review(self, recent_data, context):
    try:
        # 🔥 修复 2: 完善 L2 状态管理
        from zulong.core.state_manager import state_manager
        from zulong.core.types import L2Status
        
        # 设置 L2 状态为 REVIEW_ANALYZING
        state_manager.set_l2_status(
            L2Status.REVIEW_ANALYZING, 
            task_id=self.review_session_id
        )
        logger.info(f"✅ L2 状态已设置为 REVIEW_ANALYZING")
        
        # ... 分析、生成经验、应用经验 ...
        
        # 🔥 修复 2: 重置 L2 状态为 IDLE
        state_manager.set_l2_status(
            L2Status.IDLE, 
            task_id=self.review_session_id
        )
        logger.info(f"✅ L2 状态已重置为 IDLE")
        
    except Exception as e:
        # 🔥 修复 2: 失败时也重置 L2 状态
        state_manager.set_l2_status(L2Status.IDLE)
        logger.info(f"❌ 失败，已重置 L2 状态为 IDLE")
        raise
```

```
  ↓ (开始分析)
REVIEW_ANALYZING (ReplayIntegration 设置) 🔥 新增
  ↓ (分析完成)
IDLE (ReplayIntegration 重置) 🔥 新增
```
