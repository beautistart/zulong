```python
class DeduplicationFilter:
    """三重防重复过滤器"""
    
    def should_review(self, event_data: Dict, event_type: str) -> Tuple[bool, str]:
        """判断是否需要复盘"""
        # 1. 事件级过滤
        event_pass, reason = self._event_level_filter(event_data, event_type)
        if not event_pass:
            return False, reason
        
        # 2. 内容级过滤（向量查重）
        content_pass, reason = self._content_level_filter(event_data)
        if not content_pass:
            return False, reason
        
        # 3. 时间级过滤
        time_pass, reason = self._time_level_filter(event_data)
        if not time_pass:
            return False, reason
        
        return True, "通过所有过滤"
```

**过滤规则**:

| 过滤级别 | 规则 | 实现状态 |
|---------|------|---------|
| 事件级 | 失败事件 100% 复盘，成功事件 10% 抽样 | ✅ 已实现 |
| 内容级 | MD5 哈希 + 向量相似度 (>0.95 过滤) | ✅ 已实现 |
| 时间级 | 1 小时窗口内最多 5 个事件，间隔<5 分钟过滤 | ✅ 已实现 |

#### 1.2 内容级去重逻辑

```python
def _content_level_filter(self, event_data: Dict) -> Tuple[bool, str]:
    """内容级过滤（向量查重）"""
    # 1. 提取内容
    content = self._extract_content(event_data)
    
    # 2. 计算内容哈希
    content_hash = hashlib.md5(content.encode()).hexdigest()
    
    # 3. 检查哈希重复
    if content_hash in self._content_hashes:
        last_seen = self._content_hashes[content_hash]
        if datetime.utcnow() - last_seen < self.time_window:
            return False, f"内容重复（哈希匹配，{last_seen}）"
    
    # 4. 向量相似度检查（如果有 embedding 模型）
    if self.embedding_model:
        similar_events = self._find_similar_events(content)
        if similar_events:
            max_similarity = max(sim[1] for sim in similar_events)
            if max_similarity > self.similarity_threshold:
                return False, f"内容高度相似 (similarity={max_similarity:.3f})"
    
    return True, "内容不重复"
```

## 🚨 发现的问题

### 问题 1: 向量去重功能未实现

**位置**: `zulong/review/deduplication.py:_find_similar_events()`

**现状**: 空实现，直接返回空列表

**影响**: 
- 内容级过滤只能检测 MD5 哈希完全重复的内容
- 无法检测语义相似但表述不同的经验
- 可能导致相似经验重复入库

### 问题 2: 写入前未检索经验库

**位置**: `zulong/review/safe_applier.py:apply_experiences()`

**现状**: 直接写入，不先检索经验库

**影响**:
- 即使复盘阶段过滤了重复事件，但如果是不同会话生成的相似经验，仍会重复入库
- 经验库会随时间推移变得臃肿

### 问题 3: 缺少全局去重策略

**现状**: 
- 复盘阶段有去重过滤器（但向量搜索未实现）
- 写入阶段没有去重检查
- RAG 库没有去重逻辑

**影响**: 
- 去重逻辑不统一
- 依赖单一环节，容易漏掉重复经验

### 方案 1: 实现向量去重搜索（高优先级）

**修改文件**: `zulong/review/deduplication.py`

```python
def _find_similar_events(self, content: str, limit: int = 5):
    """查找相似事件（向量相似度）"""
    if not self.embedding_model:
        # 降级：使用关键词匹配
        return self._keyword_match_similar(content, limit)
    
    try:
        # 1. 计算 query 向量
        query_embedding = self.embedding_model.encode(content)
        
        # 2. 从经验库检索相似经验
        from zulong.replay.experience_store import get_experience_store
        store = get_experience_store()
        
        # 3. 向量相似度搜索
        similar_results = await store.search_by_embedding(
            query_embedding, 
            top_k=limit,
            min_similarity=0.7  # 降低阈值，用于去重参考
        )
        
        # 4. 转换为返回格式
        similar_events = [
            (result.content, result.similarity)
            for result in similar_results
        ]
        
        logger.debug(f"[DedupFilter] 找到{len(similar_events)}个相似事件")
        return similar_events
        
    except Exception as e:
        logger.error(f"[DedupFilter] 向量相似度搜索失败：{e}")
        return []

def _keyword_match_similar(self, content: str, limit: int = 5):
    """降级方案：关键词匹配"""
    # 从经验库获取所有经验
    from zulong.replay.experience_store import get_experience_store
    store = get_experience_store()
    
    # 关键词匹配分数
    similar_events = []
    for entry_id, entry in store._entries.items():
        score = self._compute_keyword_similarity(content, entry)
        if score > 0.5:
            similar_events.append((entry.content, score))
    
    # 按相似度排序
    similar_events.sort(key=lambda x: x[1], reverse=True)
    return similar_events[:limit]
```

### 方案 2: 写入前去重检查（高优先级）

**修改文件**: `zulong/review/safe_applier.py`

```python
async def apply_experiences(self, experiences: List[Dict], session_id: str) -> Dict:
    """应用经验到数据库（带去重检查）"""
    result = {
        'success': True,
        'applied_count': 0,
        'skipped_duplicate_count': 0,
        'failed_count': 0,
        'errors': []
    }
    
    try:
        # 1. 验证每条经验
        validated_experiences = self._validate_experiences(experiences)
        
        # 2. 批量写入（先去重）
        for exp in validated_experiences:
            try:
                # 🔥 新增：去重检查
                is_duplicate, similar_exp = await self._check_duplicate(exp)
                if is_duplicate:
                    logger.info(f"[SafeApplier] 跳过重复经验："
                               f"{exp['content'][:50]}... "
                               f"(相似度={similar_exp['similarity']:.3f})")
                    result['skipped_duplicate_count'] += 1
                    continue
                
                # 写入数据库
                self._write_experience_to_db(exp, session_id)
                result['applied_count'] += 1
            except Exception as e:
                logger.error(f"[SafeApplier] 写入经验失败：{e}")
                result['failed_count'] += 1
                result['errors'].append(str(e))
        
        if result['failed_count'] > 0:
            result['success'] = False
        
        logger.info(f"[SafeApplier] 应用完成："
                   f"{result['applied_count']}成功，"
                   f"{result['skipped_duplicate_count']}重复跳过，"
                   f"{result['failed_count']}失败")
        
    except Exception as e:
        logger.error(f"[SafeApplier] 应用经验失败：{e}", exc_info=True)
        result['success'] = False
        result['errors'].append(str(e))
    
    return result

async def _check_duplicate(self, exp: Dict) -> Tuple[bool, Optional[Dict]]:
    """检查经验是否重复
    
    Args:
        exp: 经验数据
        
    Returns:
        Tuple[bool, Optional[Dict]]: (是否重复，最相似的经验)
    """
    try:
        # 1. 获取经验库
        from zulong.replay.experience_store import get_experience_store
        store = get_experience_store()
        
        # 2. 计算向量
        content = exp['content']
        
        # 3. 搜索相似经验
        similar_results = await store.search(
            query=content,
            top_k=1,
            min_confidence=0.0  # 不限制置信度，全部检索
        )
        
        # 4. 检查相似度
        if similar_results:
            most_similar_entry, similarity = similar_results[0]
            if similarity > 0.95:  # 相似度阈值
                return True, {
                    'content': most_similar_entry.patch.condition,
                    'similarity': similarity
                }
        
        return False, None
        
    except Exception as e:
        logger.error(f"[SafeApplier] 去重检查失败：{e}")
        return False, None  # 异常时不阻止写入
```

```python
class ExperienceRAG(BaseRAGLibrary):
    def add_document(self, document: RAGDocument) -> str:
        """添加经验文档（带去重检查）"""
        # 🔥 新增：先搜索相似文档
        if document.embedding is not None:
            similar_docs = self.search_documents_by_embedding(
                document.embedding, 
                top_k=1
            )
            
            if similar_docs:
                most_similar = similar_docs[0]
                if hasattr(most_similar, 'similarity'):
                    if most_similar.similarity > 0.95:
                        # 发现重复，返回已有文档 ID
                        logger.info(f"[ExperienceRAG] 发现重复经验，"
                                   f"相似度={most_similar.similarity:.3f}")
                        # 返回已有文档 ID，或抛出异常
                        return most_similar.metadata.get('doc_id', '')
        
        # 生成文档 ID
        doc_id = f"exp_{len(self.documents)}_{int(document.created_at)}"
        
        # 存储文档
        self.documents[doc_id] = document
        
        # 添加到向量索引
        if document.embedding is not None:
            self.vector_store.add_vectors_with_ids(
                document.embedding,
                metadata=[document.to_dict()],
                vector_ids=[doc_id]
            )
        
        self.total_adds += 1
        logger.debug(f"[ExperienceRAG] Added document: {doc_id}")
        
        return doc_id
```
