### 修改 2: 实现文件打开功能

在 [`index.html#L2487`](file://d:\AI\project\zulong_beta4\openclaw_bridge\web\static\index.html#L2487-L2508) 中添加:

```javascript
// 🔥 打开文件到 IDE (通过 API 调用)
function openFileInIDE(filePath) {
    console.log('[Web] 请求打开文件:', filePath);
    
    // 方案 1: 调用后端 API (如果 Web 服务器支持)
    fetch('/api/open-file', {
        method: 'POST',
        headers: {'Content-Type': 'application/json'},
        body: JSON.stringify({path: filePath})
    }).then(function(res) {
        if (res.ok) {
            console.log('[Web] 文件已在 IDE 中打开');
        } else {
            // 降级方案：直接在新窗口打开
            window.open('file://' + filePath, '_blank');
        }
    }).catch(function(err) {
        console.warn('[Web] API 调用失败，尝试直接打开:', err);
        window.open('file://' + filePath, '_blank');
    });
}
```

**功能**:
- ✅ 优先调用后端 API 打开文件 (更安全)
- ✅ API 失败时降级为 `file://` 协议直接打开
- ✅ 支持跨浏览器兼容
