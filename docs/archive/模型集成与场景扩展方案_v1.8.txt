## 1. 模型集成方案

### 1.1 模型选型原则

根据 TSD v1.7 核心规范，系统需同时常驻三个模型，且必须使用 `Q4_K_M` 量化版以适配 RTX 3060 6GB 显存。

#### 推荐模型配置

| 模型 | 推荐型号 | 量化版本 | 显存占用 | 用途 |
|------|---------|---------|---------|------|
| **左脑** | Qwen2.5-7B-Instruct | Q4_K_M | ~4.2GB | 快速响应、简单对话 |
| **右脑** | Llama-3-8B-Instruct | Q4_K_M | ~4.8GB | 深度推理、复杂任务 |
| **测试模型** | Qwen2.5-0.5B-Instruct | Q4_K_M | ~0.5GB | 开发测试、快速验证 |
| **Embedding** | bge-m3 | FP16 (CPU) | ~500MB (内存) | 向量嵌入 |

### 1.2 模型下载脚本

创建 `scripts/download_models.py`：

```python
#!/usr/bin/env python3
"""
模型下载脚本
支持下载测试模型和生产模型
"""

import os
import sys
from huggingface_hub import snapshot_download

# 模型配置
MODELS = {
    "test": {
        "repo_id": "Qwen/Qwen2.5-0.5B-Instruct-GGUF",
        "local_dir": "models/test_model",
        "allow_patterns": ["*Q4_K_M.gguf"],
        "description": "测试模型 (0.5B)"
    },
    "left_brain": {
        "repo_id": "Qwen/Qwen2.5-7B-Instruct-GGUF",
        "local_dir": "models/left_brain",
        "allow_patterns": ["*Q4_K_M.gguf"],
        "description": "左脑模型 (7B)"
    },
    "right_brain": {
        "repo_id": "meta-llama/Llama-3-8b-instruct-GGUF",
        "local_dir": "models/right_brain",
        "allow_patterns": ["*Q4_K_M.gguf"],
        "description": "右脑模型 (8B)"
    }
}

def download_model(model_key: str):
    """下载指定模型"""
    if model_key not in MODELS:
        print(f"❌ 错误：未知模型 '{model_key}'")
        print(f"可用模型：{list(MODELS.keys())}")
        return False
    
    config = MODELS[model_key]
    
    print(f"\n📥 开始下载：{config['description']}")
    print(f"   仓库：{config['repo_id']}")
    print(f"   目录：{config['local_dir']}")
    
    try:
        # 创建本地目录
        os.makedirs(config['local_dir'], exist_ok=True)
        
        # 下载模型
        snapshot_download(
            repo_id=config['repo_id'],
            local_dir=config['local_dir'],
            allow_patterns=config['allow_patterns'],
            local_dir_use_symlinks=False,
            resume_download=True
        )
        
        print(f"✅ 下载完成：{config['description']}")
        return True
        
    except Exception as e:
        print(f"❌ 下载失败：{e}")
        return False

def main():
    """主函数"""
    if len(sys.argv) < 2:
        print("用法：python download_models.py <model_name>")
        print("\n可用模型:")
        for key, config in MODELS.items():
            print(f"  {key}: {config['description']}")
        print("\n示例:")
        print("  python download_models.py test          # 下载测试模型")
        print("  python download_models.py left_brain    # 下载左脑模型")
        print("  python download_models.py all           # 下载所有模型")
        sys.exit(1)
    
    model_name = sys.argv[1]
    
    if model_name == "all":
        # 下载所有模型
        success_count = 0
        for key in MODELS.keys():
            if download_model(key):
                success_count += 1
        print(f"\n📊 下载完成：{success_count}/{len(MODELS)} 个模型")
    else:
        # 下载单个模型
        success = download_model(model_name)
        sys.exit(0 if success else 1)

if __name__ == "__main__":
    main()
```

### 1.3 模型加载器

创建 `zulong/models/model_loader.py`：

```python
# File: zulong/models/model_loader.py
# 模型加载器 - 支持 Q4_K_M 量化版

import logging
from typing import Dict, Any, Optional
from pathlib import Path
from llama_cpp import Llama

logger = logging.getLogger(__name__)


class ModelConfig:
    """模型配置"""
    
    # 测试模型 (0.5B)
    TEST_MODEL = {
        "path": "models/test_model",
        "pattern": "*Q4_K_M.gguf",
        "n_ctx": 2048,
        "n_batch": 512,
        "n_threads": 4
    }
    
    # 左脑模型 (7B)
    LEFT_BRAIN = {
        "path": "models/left_brain",
        "pattern": "*Q4_K_M.gguf",
        "n_ctx": 4096,
        "n_batch": 512,
        "n_threads": 8
    }
    
    # 右脑模型 (8B)
    RIGHT_BRAIN = {
        "path": "models/right_brain",
        "pattern": "*Q4_K_M.gguf",
        "n_ctx": 8192,
        "n_batch": 512,
        "n_threads": 8
    }


class ModelLoader:
    """模型加载器（单例模式）"""
    
    _instance = None
    _models: Dict[str, Llama] = {}
    
    def __new__(cls):
        """单例模式"""
        if cls._instance is None:
            cls._instance = super().__new__(cls)
        return cls._instance
    
    def load_model(
        self,
        model_name: str,
        use_gpu: bool = True,
        verbose: bool = False
    ) -> Optional[Llama]:
        """加载模型
        
        Args:
            model_name: 模型名称 (test/left_brain/right_brain)
            use_gpu: 是否使用 GPU
            verbose: 是否显示详细信息
            
        Returns:
            Llama: 模型实例，加载失败返回 None
        """
        if model_name in self._models:
            logger.info(f"[ModelLoader] Model '{model_name}' already loaded")
            return self._models[model_name]
        
        # 获取配置
        config = getattr(ModelConfig, model_name.upper(), None)
        if not config:
            logger.error(f"[ModelLoader] Unknown model: {model_name}")
            return None
        
        # 查找模型文件
        model_path = self._find_model_file(config["path"], config["pattern"])
        if not model_path:
            logger.error(f"[ModelLoader] Model file not found: {config['path']}")
            return None
        
        logger.info(f"[ModelLoader] Loading {model_name} from {model_path}")
        
        try:
            # 加载模型
            model = Llama(
                model_path=model_path,
                n_ctx=config["n_ctx"],
                n_batch=config["n_batch"],
                n_threads=config["n_threads"],
                use_mmap=True,
                use_mlock=True,
                verbose=verbose
            )
            
            self._models[model_name] = model
            
            logger.info(f"[ModelLoader] Successfully loaded {model_name}")
            return model
            
        except Exception as e:
            logger.error(f"[ModelLoader] Failed to load {model_name}: {e}")
            return None
    
    def _find_model_file(self, directory: str, pattern: str) -> Optional[str]:
        """查找模型文件"""
        dir_path = Path(directory)
        if not dir_path.exists():
            return None
        
        # 查找匹配的文件
        for file in dir_path.glob(pattern):
            return str(file)
        
        return None
    
    def unload_model(self, model_name: str) -> bool:
        """卸载模型"""
        if model_name not in self._models:
            return False
        
        del self._models[model_name]
        logger.info(f"[ModelLoader] Unloaded {model_name}")
        return True
    
    def get_model(self, model_name: str) -> Optional[Llama]:
        """获取已加载的模型"""
        return self._models.get(model_name)
    
    def get_loaded_models(self) -> Dict[str, Llama]:
        """获取所有已加载的模型"""
        return self._models.copy()
    
    def get_memory_usage(self) -> Dict[str, Any]:
        """获取内存使用情况"""
        # 这里可以集成 psutil 来获取实际显存占用
        return {
            "loaded_models": len(self._models),
            "model_names": list(self._models.keys())
        }
```

## 3. 显存优化策略

### 3.1 显存预算分配（RTX 3060 6GB）

| 组件 | 显存占用 | 优化策略 |
|------|---------|---------|
| **左脑** | 4.2GB | Q4_K_M 量化 + 按需加载 |
| **右脑** | 4.8GB | Q4_K_M 量化 + 冷备模式 |
| **Embedding** | 0GB | CPU 运行 |
| **TTS** | 0GB | CPU 运行 |
| **系统预留** | 0.5GB | 操作系统和驱动 |

**总计**: 9.5GB（通过热切换机制可在 6GB 上运行）

### 3.2 热切换策略

```python
# 伪代码示例
class ModelManager:
    def __init__(self):
        self.loader = ModelLoader()
        self.active_model = None
    
    def switch_to_left_brain(self):
        """切换到左脑（快速响应）"""
        if self.active_model == "right_brain":
            self.loader.unload_model("right_brain")
        
        self.loader.load_model("left_brain")
        self.active_model = "left_brain"
    
    def switch_to_right_brain(self):
        """切换到右脑（深度推理）"""
        if self.active_model == "left_brain":
            self.loader.unload_model("left_brain")
        
        self.loader.load_model("right_brain")
        self.active_model = "right_brain"
    
    def switch_to_test_model(self):
        """切换到测试模型（开发模式）"""
        if self.active_model in ["left_brain", "right_brain"]:
            self.loader.unload_model(self.active_model)
        
        self.loader.load_model("test")
        self.active_model = "test"
```

### 3.3 显存监控

创建监控脚本 `scripts/monitor_vram.py`：

```python
#!/usr/bin/env python3
"""显存监控脚本"""

import psutil
import pynvml
import time

def init_nvml():
    """初始化 NVML"""
    pynvml.nvmlInit()
    handle = pynvml.nvmlDeviceGetHandleByIndex(0)
    return handle

def get_vram_usage(handle):
    """获取显存使用"""
    info = pynvml.nvmlDeviceGetMemoryInfo(handle)
    return {
        "total": info.total / 1024**2,  # MB
        "used": info.used / 1024**2,
        "free": info.free / 1024**2
    }

def monitor():
    """实时监控"""
    handle = init_nvml()
    
    print("📊 显存实时监控 (Ctrl+C 退出)")
    print("=" * 60)
    
    try:
        while True:
            usage = get_vram_usage(handle)
            
            print(f"总显存：{usage['total']:.0f} MB")
            print(f"已使用：{usage['used']:.0f} MB ({usage['used']/usage['total']*100:.1f}%)")
            print(f"空闲：{usage['free']:.0f} MB")
            print("-" * 60)
            
            time.sleep(2)
    
    except KeyboardInterrupt:
        print("\n👋 监控结束")
        pynvml.nvmlShutdown()

if __name__ == "__main__":
    monitor()
```

```python
# File: tests/scenario_tests/base.py
# 场景测试基类

import time
import logging
from typing import Dict, Any, List
from abc import ABC, abstractmethod

logger = logging.getLogger(__name__)


class ScenarioTest(ABC):
    """场景测试基类"""
    
    def __init__(self, name: str, description: str):
        """初始化
        
        Args:
            name: 场景名称
            description: 场景描述
        """
        self.name = name
        self.description = description
        self.results: List[Dict[str, Any]] = []
    
    @abstractmethod
    def setup(self):
        """设置测试环境"""
        pass
    
    @abstractmethod
    def run(self):
        """运行测试"""
        pass
    
    @abstractmethod
    def teardown(self):
        """清理测试环境"""
        pass
    
    def execute(self) -> bool:
        """执行完整测试流程"""
        logger.info(f"🎬 开始场景测试：{self.name}")
        
        try:
            # 设置
            self.setup()
            
            # 运行
            result = self.run()
            
            # 清理
            self.teardown()
            
            # 记录结果
            self.results.append({
                "scenario": self.name,
                "success": result,
                "timestamp": time.time()
            })
            
            return result
            
        except Exception as e:
            logger.error(f"❌ 场景测试失败：{e}")
            self.results.append({
                "scenario": self.name,
                "success": False,
                "error": str(e),
                "timestamp": time.time()
            })
            return False
    
    def get_report(self) -> Dict[str, Any]:
        """获取测试报告"""
        return {
            "name": self.name,
            "description": self.description,
            "results": self.results
        }
```
